**RademacherRVGenerator**
{% highlight C# %}
public RademacherRVGenerator()
{
    r = new Random();
}
{% endhighlight %}

{% highlight C# %}
int[] pv = {1, -1};
public int GetValue()
{
    return pv[r.Next(0,2)];
}
{% endhighlight %}

{% highlight C# %}
public double GetJump(double n)
{
    return Math.Sqrt(1 / n) * (double)GetValue();
}
{% endhighlight %}

{% highlight C# %}
public List<List<PointF>> GetJumpsPaths(int n, int m)
{
    List<List<PointF>> res = new List<List<PointF>>();

    for(int i = 0; i < m; i++)
    {
        List<PointF> path = new List<PointF>();
        double pos = 0;
        for(int j = 0; j < n; j++)
        {
            pos += GetJump(n);
            path.Add(new PointF(j, (float)pos));
        }
        res.Add(path);
    }

    return res;
}
{% endhighlight %}

**NormalRVGenerator**

{% highlight C# %}
public NormalRVGenerator()
{
    r = new Random();
}
{% endhighlight %}

{% highlight C# %}
public double GetStandardValue()
{
    double u1 = r.NextDouble();
    double u2 = r.NextDouble();
    return Math.Sqrt(-2 * Math.Log(u1))*Math.Cos(2*Math.PI*u2);
}                    
{% endhighlight %}

{% highlight C# %}
public List<List<PointF>> GetJumpsPaths(int n, int m)
{
    List<List<PointF>> res = new List<List<PointF>>();

    for (int i = 0; i < m; i++)
    {
        List<PointF> path = new List<PointF>();
        double pos = 0;
        for (int j = 0; j < n; j++)
        {
            pos += GetStandardValue();
            path.Add(new PointF(j, (float)pos));
        }
        res.Add(path);
    }

    return res;
}
